### They replace old districts with new good ones lol
namespace = aot_ai_districts

country_event = {
    id = aot_ai_districts.1
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        is_ai = yes
        OR = {
            has_dm_capital_tech = yes
        }
    }
    immediate = {
        every_owned_planet = {
            limit = {
                is_exclusively_delta_tier = yes
                has_any_outdated_zone = yes
            }

            # Upgrade all non urban zones.
            bulk_upgrade_zones = { TYPE_STANDARD = acot_dm TYPE_SPECIAL = aot_DM }

            # Upgrade urban zones now.
            replace_urban_zones = { DISTRICT = district_city ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_hive ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_nexus ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_rw_city ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_arcology_housing ZONE = zone_acot_dm_urban }
        }
    }
}

# Upgrade capital buildings when eligible, the AI likes to eat glue way too much.
country_event = {
    id = aot_ai_districts.2
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        is_ai = yes
        OR = {
            has_technology = tech_acot_dm_ancient_palace
            has_technology = tech_acot_dm_ancient_control_center
            has_technology = tech_acot_dm_ancient_hive_cluster
        }
    }
    immediate = {
        every_owned_planet = {
            limit = {
                has_enigmatic_capital = no
            }
            set_update_modifiers_batch = begin
            hidden_effect = {
                # AI got enough brains now to do the superior constructs decision but the extra giga fortress is still nice tho.
                # remove_capital_building = yes
                # add_advanced_capital_building = yes
                if = {
                    limit = {
                        owner = {
                            has_technology = tech_giga_fortress_dm
                        }
                        NOT = {
                            has_building = building_giga_fortress_dm
                        }
                    }
                    # FREE GIGAFORT BABY
                    add_building = building_giga_fortress_dm
                }
                # Add conduit to planet if there's an AoT dark matter habitat since this is important.
                if = {
                    limit = {
                        this = {
                            is_aot_habitat = no
                        }
                        owner = {
                            has_technology = tech_delta_nexus
                        }
                        solar_system = {
                            any_system_planet = {
                                is_aot_habitat = yes
                            }
                        }
                        NOT = {
                            has_building = building_delta_nexus
                        }
                    }
                    # FREE GIGAFORT BABY
                    add_building = building_delta_nexus
                }
            }
            set_update_modifiers_batch = end
        }
    }
}

# Change planets into fractured worlds, to prepare for voidspheres I guess.
country_event = {
    id = aot_ai_districts.8000
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        NOT = {
            has_country_flag = free_fractured_world
        }
        is_ai = yes
        has_technology = tech_bombardment_dm
        any_planet_within_border = {
            OR = {
                is_planet_class = pc_barren
                is_planet_class = pc_barren_cold
                is_planet_class = pc_frozen
                is_planet_class = pc_toxic
                is_planet_class = pc_molten
            }
        }
    }
    immediate = {
        random_planet_within_border = {
            limit = {
                OR = {
                    is_planet_class = pc_barren
                    is_planet_class = pc_barren_cold
                    is_planet_class = pc_frozen
                    is_planet_class = pc_toxic
                    is_planet_class = pc_molten
                }
            }
            change_pc = pc_dark_fractured
            #add_modifier = {
            #	modifier = acot_modifier_precursor_digsite
            #	days = -1
            #}
            set_planet_flag = free_fractured_planet
            planet_event = {
                id = acot_exterminatus_event.1000
            }
            random_list = {
                75 = {
                    add_deposit = d_acot_void_chasm
                }
                15 = {
                    add_deposit = d_acot_void_chasm
                    add_deposit = d_acot_void_rift
                }
                10 = {
                    add_deposit = d_acot_void_chasm
                    add_deposit = d_acot_void_rift
                    add_deposit = d_acot_void_terrain
                }
            }
        }
        set_country_flag = free_fractured_world
    }
}

### Help them along

country_event = {
    id = aot_ai_districts.5
    is_triggered_only = yes
    hide_window = yes
    trigger = {
        is_ai = yes
        any_planet_within_border = {
            is_planet_class = pc_acot_pmc_vault
            NOT = {
                has_planet_flag = reinforce_defense_armies
            }
            solar_system = {
                NOT = {
                    any_ship_in_system = {
                        owner = {
                            is_country_type = aot_phanon_protecc
                        }
                    }
                }
            }
        }
    }
    immediate = {
        every_planet_within_border = {
            limit = {
                is_planet_class = pc_acot_pmc_vault
            }
            root = {
                fire_on_action = {
                    on_action = on_planet_attackers_win
                    scopes = {
                        from = home_planet.owner
                        fromfrom = prev
                    }
                }
            }
            set_owner = root
            set_planet_flag = reinforce_defense_armies
            planet_event = {
                id = aot_ai_districts.52
                days = 30
            }
        }
    }
}

planet_event = {
    id = aot_ai_districts.52
    is_triggered_only = yes
    hide_window = yes
    immediate = {
        if = {
            limit = {
                exists = space_owner
                exists = owner
                space_owner = {
                    is_same_value = owner
                }
                has_planet_flag = reinforce_defense_armies
            }
            add_modifier = {
                modifier = acot_modifier_defense_upgrade_2
                days = -1
            }
            while = {
                count = 400
                create_army = {
                    owner = space_owner
                    type = acot_defense_army_heavy_ae
                }
            }
        }
    }
}
