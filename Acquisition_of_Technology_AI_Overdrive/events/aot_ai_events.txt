namespace = aot_ai_districts

country_event = {
    id = aot_ai_districts.1
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        is_ai = yes
        OR = {
            has_dm_capital_tech = yes
        }
    }
    immediate = {
        every_owned_planet = {
            limit = {
                is_exclusively_delta_tier = yes
                has_any_outdated_zone = yes
            }

            # Upgrade all non urban zones.
            bulk_upgrade_zones = { TYPE_STANDARD = acot_dm TYPE_SPECIAL = aot_DM }

            # Upgrade urban zones now.
            replace_urban_zones = { DISTRICT = district_city ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_hive ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_nexus ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_rw_city ZONE = zone_acot_dm_urban }
            replace_urban_zones = { DISTRICT = district_arcology_housing ZONE = zone_acot_dm_urban }
        }
    }
}

# Watch over the AI if they have capacity to build new labs but somehow refuse to build them.
country_event = {
    id = aot_ai_districts.4
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        is_ai = yes
        OR = {
            has_monthly_income = {
                resource = society_research
                value < value:expected_research_value_empire
            }
            has_monthly_income = {
                resource = physics_research
                value < value:expected_research_value_empire
            }
            has_monthly_income = {
                resource = engineering_research
                value < value:expected_research_value_empire
            }
        }
        if = {
            limit = { country_uses_consumer_goods = yes }
            NOT = {
                has_deficit = consumer_goods
            }
        }
        else = {
            NOT = {
                has_deficit = energy
            }
        }
        any_owned_planet = {
            free_jobs > 0
            switch = {
                trigger = is_planet_class
                pc_ringworld_habitable = {
                    free_district_slots > 0
                }
                pc_city = {
                    free_district_slots > 0
                }
                pc_machine = {
                    free_district_slots > 0
                }
                pc_hive = {
                    free_district_slots > 0
                }
                # Default case for any other planet out there
                default = {
                    free_building_slots > 0
                }
            }

        }
    }
    immediate = {
        random_owned_planet = {
            limit = {
                OR = {
                    has_designation = col_city
                    has_designation = col_research
                    has_designation = col_ecu_research
                    has_designation = col_ring_research
                    has_designation = col_habitat_research
                }
                free_jobs > 0
                switch = {
                    trigger = is_planet_class
                    pc_ringworld_habitable = {
                        free_district_slots > 0
                    }
                    pc_city = {
                        free_district_slots > 0
                    }
                    pc_machine = {
                        free_district_slots > 0
                    }
                    pc_hive = {
                        free_district_slots > 0
                    }
                    # Default case for any other planet out there
                    default = {
                        free_building_slots > 0
                    }
                }

            }
            weights = {
                base = 10
                modifier = {
                    factor = 4
                    has_designation = col_research
                }
                modifier = {
                    factor = 6
                    has_designation = col_habitat_research
                }
                modifier = {
                    factor = 8
                    has_designation = col_ecu_research
                }
                modifier = {
                    factor = 10
                    has_designation = col_ring_research
                }
            }
            # Add the lab of which we have the least monthly tech for
            switch = {
                trigger = is_planet_class
                pc_ringworld_habitable = {
                    tiered_generic_special_zone_replace = { DISTRICT = district_rw_urban_3 TYPE = research PLANET_TYPE = ring_world }
                    add_district_if_possible_valid_district_effect = { DISTRICT = district_rw_urban_3 }
                }
                pc_city = {
                    tiered_generic_special_zone_replace = { DISTRICT = district_arcology_urban_3 TYPE = research PLANET_TYPE = arcology }
                    add_district_if_possible_valid_district_effect = { DISTRICT = district_arcology_urban_3 }
                }
                pc_machine = {
                    tiered_generic_special_zone_replace = { DISTRICT = district_nexus_3 TYPE = research PLANET_TYPE = nexus }
                    add_district_if_possible_valid_district_effect = { DISTRICT = district_nexus_3 }
                }
                pc_hive = {
                    tiered_generic_special_zone_replace = { DISTRICT = district_hive_3 TYPE = research PLANET_TYPE = hive }
                    add_district_if_possible_valid_district_effect = { DISTRICT = district_hive_3 }
                }
                # Default case for any other planet out there
                default = {
                    # Do the actual business now
                    owner = {
                        export_trigger_value_to_variable = {
                            trigger = has_monthly_income
                            parameters = {
                                resource = physics_research
                            }
                            variable = physics_research_income_current
                        }
                        export_trigger_value_to_variable = {
                            trigger = has_monthly_income
                            parameters = {
                                resource = society_research
                            }
                            variable = society_research_income_current
                        }
                        export_trigger_value_to_variable = {
                            trigger = has_monthly_income
                            parameters = {
                                resource = engineering_research
                            }
                            variable = engineering_research_income_current
                        }
                    }
                    if = {
                        limit = { has_enigmatic_capital = yes }
                        add_building = building_acot_dm_enigmatic_lab
                    }
                    else_if = {
                        limit = {
                            owner = {
                                has_monthly_income = { resource = physics_research value < society_research_income_current }
                                has_monthly_income = { resource = physics_research value < engineering_research_income_current }
                            }
                        }
                        add_building = building_physics_lab_1
                    }
                    else_if = {
                        limit = {
                            owner = {
                                has_monthly_income = { resource = society_research value < physics_research_income_current }
                                has_monthly_income = { resource = society_research value < engineering_research_income_current }
                            }
                        }
                        add_building = building_biolab_1
                    }
                    else = {
                        add_building = building_engineering_facility_1
                    }
                    # Cleanup variables.
                    clear_variable = physics_research_income_current
                    clear_variable = society_research_income_current
                    clear_variable = engineering_research_income_current
                }
            }
        }
    }
}

# Change planets into fractured worlds, to prepare for voidspheres I guess.
country_event = {
    id = aot_ai_districts.8000
    hide_window = yes
    is_triggered_only = yes
    trigger = {
        NOT = {
            has_country_flag = free_fractured_world
        }
        is_ai = yes
        has_technology = tech_bombardment_dm
        any_planet_within_border = {
            OR = {
                is_planet_class = pc_barren
                is_planet_class = pc_barren_cold
                is_planet_class = pc_frozen
                is_planet_class = pc_toxic
                is_planet_class = pc_molten
            }
        }
    }
    immediate = {
        random_planet_within_border = {
            limit = {
                OR = {
                    is_planet_class = pc_barren
                    is_planet_class = pc_barren_cold
                    is_planet_class = pc_frozen
                    is_planet_class = pc_toxic
                    is_planet_class = pc_molten
                }
            }
            change_pc = pc_dark_fractured
            #add_modifier = {
            #	modifier = acot_modifier_precursor_digsite
            #	days = -1
            #}
            set_planet_flag = free_fractured_planet
            planet_event = {
                id = acot_exterminatus_event.1000
            }
            random_list = {
                75 = {
                    add_deposit = d_acot_void_chasm
                }
                15 = {
                    add_deposit = d_acot_void_chasm
                    add_deposit = d_acot_void_rift
                }
                10 = {
                    add_deposit = d_acot_void_chasm
                    add_deposit = d_acot_void_rift
                    add_deposit = d_acot_void_terrain
                }
            }
        }
        set_country_flag = free_fractured_world
    }
}

### Help them along

country_event = {
    id = aot_ai_districts.5
    is_triggered_only = yes
    hide_window = yes
    trigger = {
        is_ai = yes
        any_planet_within_border = {
            is_planet_class = pc_acot_pmc_vault
            NOT = {
                has_planet_flag = reinforce_defense_armies
            }
            solar_system = {
                NOT = {
                    any_ship_in_system = {
                        owner = {
                            is_country_type = aot_phanon_protecc
                        }
                    }
                }
            }
        }
    }
    immediate = {
        every_planet_within_border = {
            limit = {
                is_planet_class = pc_acot_pmc_vault
            }
            root = {
                fire_on_action = {
                    on_action = on_planet_attackers_win
                    scopes = {
                        from = home_planet.owner
                        fromfrom = prev
                    }
                }
            }
            set_owner = root
            set_planet_flag = reinforce_defense_armies
            planet_event = {
                id = aot_ai_districts.52
                days = 30
            }
        }
    }
}

planet_event = {
    id = aot_ai_districts.52
    is_triggered_only = yes
    hide_window = yes
    immediate = {
        if = {
            limit = {
                exists = space_owner
                exists = owner
                space_owner = {
                    is_same_value = owner
                }
                has_planet_flag = reinforce_defense_armies
            }
            add_modifier = {
                modifier = acot_modifier_defense_upgrade_2
                days = -1
            }
            while = {
                count = 400
                create_army = {
                    owner = space_owner
                    type = acot_defense_army_heavy_ae
                }
            }
        }
    }
}
