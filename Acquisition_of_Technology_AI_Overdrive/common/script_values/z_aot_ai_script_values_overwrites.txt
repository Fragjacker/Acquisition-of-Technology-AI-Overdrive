# We need more constructor ships for them crazy AI bois.
desired_constructors_min = {
    complex_trigger_modifier = {
        trigger = num_owned_planets
        trigger_scope = owner
        mode = add
    }
    divide = 10
    modifier = {
        add = 1
        OR = {
            is_country_type = default
            is_country_type = awakened_fallen_empire
        }
    }
    modifier = {
        add = 2
        is_country_type = default
        controlled_systems > 3
        years_passed < 100
    }
    round = yes
}

# This is used to count vanilla buildings such as forges, factories etc.
count_special_buildings_on_planet = {
    complex_trigger_modifier = {
        trigger = num_buildings
        parameters = {
            type = building_$BUILDING_TYPE$_1
        }
        trigger_scope = planet
        mode = add
    }
    complex_trigger_modifier = {
        trigger = num_buildings
        parameters = {
            type = building_$BUILDING_TYPE$_2
        }
        trigger_scope = planet
        mode = add
    }
    complex_trigger_modifier = {
        trigger = num_buildings
        parameters = {
            type = building_$BUILDING_TYPE$_3
        }
        trigger_scope = planet
        mode = add
    }
}
