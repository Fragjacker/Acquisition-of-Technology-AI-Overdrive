ap_world_shaper = {
    possible = {
        custom_tooltip = {
            fail_text = "requires_technology_tech_climate_restoration"
            has_technology = tech_climate_restoration
        }
        custom_tooltip = {
            fail_text = "requires_ascension_perks_1"
            num_ascension_perks > 0
        }
    }

    modifier = {
        terraforming_cost_mult = -0.25
        add_attunement_the_composer_of_strands = @ascension_perk_attunement1
    }

    on_enabled = {
        custom_tooltip = "allow_gaia"
    }

    potential = {
        NOT = {
            has_ascension_perk = ap_world_shaper
        }
        OR = {
            is_hive_empire = no
            is_wilderness_empire = yes
        }
        OR = {
            is_machine_empire = no
            has_valid_civic = civic_machine_servitor
            has_valid_civic = civic_machine_assimilator
        }
    }

    ai_weight = {
        weight = 1000
        modifier = {
            factor = 0
            OR = {
                is_void_dweller_empire = yes
                has_origin = origin_shattered_ring
                has_trait = trait_aquatic
                has_ethic = ethic_gestalt_consciousness
            }
        }
        modifier = {
            factor = 2
            OR = {
                has_ethic = ethic_pacifist
                has_ethic = ethic_xenophobe
            }
        }
        modifier = {
            factor = 5
            OR = {
                has_ethic = ethic_fanatic_pacifist
                has_ethic = ethic_fanatic_xenophobe
                is_wilderness_empire = yes
            }
        }
        modifier = {
            #only way to get more nice planets
            factor = 10
            has_origin = origin_life_seeded
        }
    }
}

ap_galactic_wonders_utopia = {
    on_enabled = {
        custom_tooltip = "add_research_options_galactic_wonders_utopia"
        hidden_effect = {
            if = {
                limit = {
                    NOT = { has_technology = tech_ring_world }
                    can_research_technology = tech_ring_world
                }
                add_research_option = tech_ring_world
            }
            if = {
                limit = {
                    NOT = { has_technology = tech_dyson_sphere }
                }
                add_research_option = tech_dyson_sphere
            }
            if = {
                limit = {
                    is_ai = yes
                    NOT = { has_technology = tech_mega_engineering }
                }
                add_research_option = tech_mega_engineering
            }
        }
    }

    possible = {
        # change to allow the AI an easier time catching up
        custom_tooltip = {
            fail_text = "requires_mega_engineering"
            OR = {
                has_technology = tech_mega_engineering
                AND = {
                    is_ai = yes
                    num_ascension_perks >= 4
                    any_country = {
                        is_ai = no
                        has_galactic_wonders = yes
                    }
                }
            }
        }
    }

    potential = {
        host_has_dlc = "Utopia"
        NOR = {
            host_has_dlc = "Megacorp"
            has_ascension_perk = ap_galactic_wonders_utopia
        }
    }

    ai_weight = {
        weight = 10000 # So much ACoT content is behind this, if you don't have it then what are you even doing?
    }
}

        modifier = {
