### im removing most of the weird little mechanics from these jobs. theyre obtuse and very hard for the AI to do.

acot_dark_energy_man = {
    category = specialist

    swappable_data = {
        default = {
            condition_string = SPECIALIST_JOB_TRIGGER
            building_icon = building_acot_ae_dark_energy_pylon
        }
    }

    tags = { strategic_resource }


    possible_pre_triggers = {
        has_owner = yes
        is_being_purged = no
        is_being_assimilated = no
        is_sapient = yes
    }

    possible_precalc = can_fill_specialist_job
    possible = {
        complex_specialist_job_check_trigger = yes
    }

    resources = {
        category = planet_jobs_productive
        produces = {
            acot_sr_dark_energy = 6
        }
        upkeep = {
            energy = 12
        }
    }

    weight = {
        weight = @specialist_job_weight
        mult = value:job_weights_modifier|JOB|acot_dark_energy_man|RESOURCE|acot_sr_dark_energy|
        modifier = {
            factor = 0.2
            has_citizenship_rights = no
            NOT = { has_trait = trait_mechanical }
        }
        modifier = {
            factor = 0.1
            can_take_servant_job = yes
        }
        modifier = {
            factor = 1.5
            from = {
                any_owned_pop_job = {
                    has_job_type = acot_dark_energy_man
                }
            }
        }
        modifier = {
            factor = 1.2
            is_enslaved = yes
            has_slavery_type = { type = slavery_indentured }
        }
    }
    promotion = {
        time = @standard_promotion_time
        effect = {
            pop_amount_change_category = {
                CATEGORY = specialist
                AMOUNT = local_pop_amount
            }
        }
    }

}
