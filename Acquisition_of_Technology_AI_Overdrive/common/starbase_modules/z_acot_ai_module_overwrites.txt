acot_precursor_starbase_shipyard = {
    icon = "GFX_spaceport_module_acot_precursor_starbase_shipyard"
    section = "ACOT_SHIPYARD_STARBASE_SECTION"
    construction_days = 5
    initial = yes
    destroy_trigger = {
        has_starbase_size > acot_precusor_starcitadel
    }
    potential = {
        has_starbase_size >= acot_precusor_starbase
        has_starbase_size < acot_precusor_phanon_starcitadel
    }

    resources = {
        category = starbase_modules
        cost = {
        }
        upkeep = {
            trigger = {
                solar_system = {
                    NOT = {
                        is_star_class = sc_black_hole
                    }

                }
            }
            #acot_sr_dark_energy = 5
            #sr_dark_matter = 5
        }
    }

    station_modifier = {
        starbase_shipyard_capacity_add = 5
    }
    on_finished = {
        set_ship_construction_type = starbase_shipyard
    }
    on_destroyed = {
        if = {
            limit = {
                acot_has_traditional_shipyard = no
            }
            set_ship_construction_type = none
        }
    }

    ai_build_at_chokepoint = yes
    ai_build_outside_chokepoint = yes
    ai_weight = {
        weight = 100 #use same weight here - ai weighting is done in starbase_types.txt
        modifier = {
            factor = 100
            owner = {
                OR = {
                    count_owned_starbase = {
                        count = 0
                        limit = { has_starbase_module = acot_precursor_starbase_shipyard }
                    }
                    AND = {
                        controlled_systems > 30
                        count_owned_starbase = {
                            count < 2
                            limit = { has_starbase_module = acot_precursor_starbase_shipyard }
                        }
                    }
                    AND = {
                        controlled_systems > 60
                        count_owned_starbase = {
                            count < 3
                            limit = { has_starbase_module = acot_precursor_starbase_shipyard }
                        }
                    }
                }
            }
        }
        modifier = {
            factor = 3
            has_starbase_module = acot_precursor_starbase_shipyard
            count_starbase_modules = {
                count < 2
                type = acot_precursor_starbase_shipyard
            }
        }
        modifier = {
            factor = 3
            has_starbase_module = acot_precursor_starbase_shipyard
            count_starbase_modules = {
                count < 4
                type = acot_precursor_starbase_shipyard
            }
        }
    }
}
