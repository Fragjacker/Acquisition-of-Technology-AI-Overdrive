# Replaces a single zone according to the proper tech tier.
replace_single_zone = {
    if = {
        limit = {
            exists = root
            num_districts = { type = $DISTRICT$ value > 0 } # Only add zones when at least one district has been built
        }
        add_zone = {
            district = $DISTRICT$
            zone = zone_$PREFIX$_$TYPE$
            zone_slot = 0
            replace = yes
        }
    }
}

replace_urban_zones = {
    add_zone = {
        district = $DISTRICT$
        zone = $ZONE$
        zone_slot = 1
        replace = yes
    }
    add_zone = {
        district = $DISTRICT$
        zone = $ZONE$
        zone_slot = 2
        replace = yes
    }
}

acot_ai_void_sphere_effect = {
    export_trigger_value_to_variable = {
        trigger = planet_size
        variable = district_count
    }
    multiply_variable = {
        which = district_count
        value = 1.5
    }
    change_pc = pc_acot_enigmopolis
    while = {
        count = district_count
        random_list = {
            1 = {
                add_district = district_acot_enigmapolis_city
            }
            5 = {
                add_district = district_acot_enigmapolis_science
            }
            2 = {
                add_district = district_acot_enigmapolis_forge
            }
            5 = {
                add_district = district_acot_enigmapolis_silo
            }
            1 = {
                add_district = district_acot_enigmapolis_fortress
            }
        }
    }
}

add_district_if_possible_valid_district_effect = {
    optimize_memory
    if = {
        limit = {
            has_district = $DISTRICT$
            free_district_slots > 0
        }
        add_district = $DISTRICT$
    }
    #else do nothing
}

# This fills all free districts up with the desired district type.
fill_districts_to_max = {
    while = {
        count = 100
        add_district = $DISTRICT$
    }
}
