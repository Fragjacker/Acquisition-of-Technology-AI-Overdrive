add_ai_zone_effect = {
    if = {
        limit = {
            exists = owner
            num_districts = { type = $DISTRICT$ value > 0 }
            owner = {
                has_exclusively_delta_capital_tech = yes
            }
        }
        add_zone = {
            district = $DISTRICT$
            zone = zone_$PREFIX$_$TYPE$
            zone_slot = 0
            replace = yes
        }
    }
    if = {
        limit = {
            exists = owner
            num_districts = { type = $DISTRICT$ value > 0 }
            owner = {
                has_exclusively_alpha_capital_tech = yes
            }
        }
        add_zone = {
            district = $DISTRICT$
            zone = zone_$PREFIX$_$TYPE$
            zone_slot = 0
            replace = yes
        }
    }
    if = {
        limit = {
            exists = owner
            num_districts = { type = $DISTRICT$ value > 0 }
            owner = {
                has_exclusively_phanon_capital_tech = yes
            }
        }
        add_zone = {
            district = $DISTRICT$
            zone = zone_aot_PE_$TYPE$
            zone_slot = 0
            replace = yes
        }
    }
    if = {
        limit = {
            exists = owner
            num_districts = { type = $DISTRICT$ value > 0 }
            owner = {
                has_exclusively_stellarite_capital_tech = yes
            }
        }
        add_zone = {
            district = $DISTRICT$
            zone = zone_aot_SE_$TYPE$
            zone_slot = 0
            replace = yes
        }
    }
}

replace_single_zone = {
    if = {
        limit = {
            has_district = $DISTRICT$
            num_districts = { type = $DISTRICT$ value > 0 } # Only add zones when at least one district has been built
        }
        add_ai_zone_effect = { DISTRICT = $DISTRICT$ TYPE = $TYPE$ PREFIX = $PREFIX$ }
    }
}

replace_urban_districts = {
    if = {
        limit = {
            has_district = $DISTRICT$
        }
        add_zone = {
            district = $DISTRICT$
            zone = zone_$PREFIX$_urban
            zone_slot = 1
        }
        add_zone = {
            district = $DISTRICT$
            zone = zone_$PREFIX$_urban
            zone_slot = 2
        }
    }
}

acot_ai_void_sphere_effect = {
    export_trigger_value_to_variable = {
        trigger = planet_size
        variable = district_count
    }
    multiply_variable = {
        which = district_count
        value = 1.5
    }
    change_pc = pc_acot_enigmopolis
    while = {
        count = district_count
        random_list = {
            1 = {
                add_district = district_acot_enigmapolis_city
            }
            5 = {
                add_district = district_acot_enigmapolis_science
            }
            2 = {
                add_district = district_acot_enigmapolis_forge
            }
            5 = {
                add_district = district_acot_enigmapolis_silo
            }
            1 = {
                add_district = district_acot_enigmapolis_fortress
            }
        }
    }
}
