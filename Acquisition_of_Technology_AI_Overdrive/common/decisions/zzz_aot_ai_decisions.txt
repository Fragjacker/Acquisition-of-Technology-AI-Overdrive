                modifier = acot_modifier_vault_delay

                    NOT = {
                        has_global_flag = acot_phanon_rival_plan_initiated

                    limit = {

                        id = acot_pmc.12
                    }
                }
                if = {
                    limit = {
                        planet_owner = {
                            has_country_flag = acot_phanon_tech_stage_one_enabled
                            NOT = {
                                has_country_flag = acot_phanon_tech_stage_two_enabled
                            }

                    planet_event = {

                    }
                }
                if = {
                    limit = {
                        planet_owner = {
                            has_country_flag = acot_phanon_tech_stage_two_enabled
                            NOT = {
                                has_country_flag = acot_phanon_tech_stage_three_enabled
                            }
                        }

                        id = acot_pmc.14
                    }
                }
                if = {
                    limit = {
                        planet_owner = {
                            has_country_flag = acot_phanon_tech_stage_three_enabled
                            NOT = {
                                has_country_flag = acot_phanon_tech_stage_four_enabled

# Fully commit a planet to its designation once it's been filled. Energy planets should have all their generator districts filled etc.
decision_pivot_planet_district = {
    owned_planets_only = yes
    enactment_time = 90
    resources = {
        category = decisions
        cost = {
            unity = 100
            minerals = 500
        }
    }
    potential = {
        exists = owner
        owner = {
            is_ai = yes
        }
        OR = {
            has_district = district_city
            has_district = district_hive
            has_district = district_nexus
        }
        OR = {
            has_district = district_farming
            has_district = district_mining
            has_district = district_mining_uncapped
            has_district = district_generator
            has_district = district_generator_uncapped
            has_district = district_farming
            has_district = district_farming_uncapped
        }
        has_enigmatic_capital = no  # Only for pleb tier stuffs
    }
    allow = {
        free_district_slots < 1
    }
    effect = {
        hidden_effect = {
            # Nuke all districts first, we redistribute them next.
            remove_all_districts = yes
            switch = {
                trigger = has_designation
                col_farming = {
                    # Fill up the target district up first
                    if = {
                        limit = { is_farming_district_uncapped = yes }
                        fill_districts_to_max = { DISTRICT = district_farming_uncapped }
                    }
                    else = {
                        fill_districts_to_max = { DISTRICT = district_farming }
                    }
                }
                col_mining = {
                    # Fill up the target district up first
                    if = {
                        limit = { is_mining_district_uncapped = yes }
                        fill_districts_to_max = { DISTRICT = district_mining_uncapped }
                    }
                    else = {
                        fill_districts_to_max = { DISTRICT = district_mining }
                    }
                }
                col_generator = {
                    # Fill up the target district up first
                    if = {
                        limit = { is_generator_district_uncapped = yes }
                        fill_districts_to_max = { DISTRICT = district_generator_uncapped }
                    }
                    else = {
                        fill_districts_to_max = { DISTRICT = district_generator }
                    }
                }
                # Change zones to be uniform now
                col_foundry = {
                    replace_urban_zones = { DISTRICT = district_city ZONE = zone_foundry }
                    replace_urban_zones = { DISTRICT = district_hive ZONE = zone_foundry }
                    replace_urban_zones = { DISTRICT = district_nexus ZONE = zone_foundry }
                }
                col_industrial = {
                    replace_urban_zones = { DISTRICT = district_city ZONE = zone_industrial }
                    replace_urban_zones = { DISTRICT = district_hive ZONE = zone_industrial }
                    replace_urban_zones = { DISTRICT = district_nexus ZONE = zone_industrial }
                }
                col_factory = {
                    replace_urban_zones = { DISTRICT = district_city ZONE = zone_factory }
                    replace_urban_zones = { DISTRICT = district_hive ZONE = zone_factory }
                    replace_urban_zones = { DISTRICT = district_nexus ZONE = zone_factory }
                }
                col_research = {
                    replace_urban_zones = { DISTRICT = district_city ZONE = zone_research }
                    replace_urban_zones = { DISTRICT = district_hive ZONE = zone_research }
                    replace_urban_zones = { DISTRICT = district_nexus ZONE = zone_research }
                }
            }
            ## Fill the rest up with city districts.
            if = { limit = { has_district = district_city } fill_districts_to_max = { DISTRICT = district_city } }
            else_if = { limit = { has_district = district_hive } fill_districts_to_max = { DISTRICT = district_hive } }
            else_if = { limit = { has_district = district_nexus } fill_districts_to_max = { DISTRICT = district_nexus } }
        }
    }
    ai_weight = {
        weight = 0
        modifier = {
            add = 1000
            OR = {
                has_designation = col_factory
                has_designation = col_industrial
                has_designation = col_foundry
                has_designation = col_research
            }
            OR = {
                num_districts = { type = district_farming value > 0 }
                num_districts = { type = district_farming_uncapped value > 0 }
                num_districts = { type = district_mining value > 0 }
                num_districts = { type = district_mining_uncapped value > 0 }
                num_districts = { type = district_generator value > 0 }
                num_districts = { type = district_generator_uncapped value > 0 }
            }
        }
        modifier = {
            add = 1000
            has_designation = col_generator
            OR = {
                num_districts = { type = district_mining value > 0 }
                num_districts = { type = district_mining_uncapped value > 0 }
                num_districts = { type = district_generator value > 0 }
                num_districts = { type = district_generator_uncapped value > 0 }
            }
        }
        modifier = {
            add = 1000
            has_designation = col_mining
            OR = {
                num_districts = { type = district_farming value > 0 }
                num_districts = { type = district_farming_uncapped value > 0 }
                num_districts = { type = district_generator value > 0 }
