						is_planet_class = pc_acot_enigmopolis
					}
				}
				while = {
					count = 4
					random_pop = {
						resettle_pop = {
							pop = this
							planet = root
						}
					}
				}
			}
		}
		set_update_modifiers_batch = end
	}
	ai_weight = { ### we REALLY want to max this planet out

		modifier = {
			factor = 10
			num_free_districts = {
				type = any
				value < 2

		}
	}

decision_enigmatic_mass_recruitment = {
	enactment_time = 120

	resources = {
		category = armies
		cost = {
			energy = 100
			alloys = 2500

		}
	}

	potential = {
		exists = owner
		owner = {
			is_ai = yes
			has_technology = tech_dark_matter_power_core_ae
			is_at_war = yes
		}
	}
	abort_trigger = {
		exists = owner
		owner = {
			is_at_war = no
		}
	}

	effect = {
		#custom_tooltip = decision_maginot_mass_recruitment_effect
		hidden_effect = {
			if = {
				limit = { exists = planet_owner }
				from = {
					set_country_flag = acot_decision_army_haver
				}
				if = {
					limit = { planet_owner = { is_machine_empire = no } }
					while = {
						set_update_modifiers_batch = begin
						count = 10
						create_army = {
							owner = planet_owner
							species = planet_owner.species
							type = acot_special_force_ae
							effect = {
								set_army_flag = acot_war_army
							}
						}
						set_update_modifiers_batch = end
					}
				}
				else_if = {
					limit = { planet_owner = { is_hive_empire = yes } }
					while = {
						count = 10
						create_army = {
							owner = planet_owner
							species = planet_owner.species
							type = acot_hive_heavy_ae
							effect = {
								set_army_flag = acot_war_army
							}
						}
					}
				}
				else_if = {
					limit = { planet_owner = { is_machine_empire = yes } }
					while = {
						count = 10
						create_army = {
							owner = planet_owner
							species = planet_owner.species
							type = acot_army_machine_heavy_ae
							effect = {
								set_army_flag = acot_war_army
							}
						}
					}
				}
			}
		}
	}

	ai_weight = { # this is for player  (in)convenience
		weight = 300
	}
}
