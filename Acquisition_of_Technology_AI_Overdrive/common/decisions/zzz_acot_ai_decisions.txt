decision_acot_infiltrate_phanon_grid = {
    owned_planets_only = yes
    sound = event_sensor_ping
    resources = {
        category = decisions
        cost = {
        }
    }
    allow = {
        any_planet_army = {
            is_owned_by = root.owner
        }
        NOR = {
            has_modifier = acot_modifier_vault_delay
            has_global_flag = aot_phanon_content_OFF
            has_planet_flag = ongoing_vault_infiltration
        }
        planet_owner = {
            has_technology = tech_precursor_phanon_design_pe_1
            has_technology = tech_precursor_phanon_design_pe_2
            has_technology = tech_precursor_phanon_design_pe_3
        }
        custom_tooltip = {
            fail_text = acot_infiltrate_phanon_grid_fail_text
            sapient_pop_amount >= 3000
        }
    }
    potential = {
        NOT = {
            has_global_flag = acot_phanon_rival_plan_initiated
        }
        planet_owner = {
            is_ai = no
            NOT = {
                has_country_flag = acot_phanon_tech_stage_four_enabled
            }
        }
        OR = {
            is_planet_class = pc_acot_pmc_vault
            is_same_value = event_target:acot_phanon_vault
            is_same_value = event_target:aot_bootleg_phanon_vault
        }
        NOR = {
            has_planet_flag = acot_vault_floor_four_cleared
            has_planet_flag = acot_vault_rival_cleared
        }
        has_ground_combat = no
    }
    effect = {
        hidden_effect = {
            set_planet_flag = ongoing_vault_infiltration
            add_modifier = {
                modifier = acot_modifier_vault_delay
                months = 6
            }
            if = {
                limit = {
                    NOT = {
                        has_global_flag = acot_phanon_rival_plan_initiated
                    }
                }
                if = {
                    limit = {
                        planet_owner = {
                            NOT = {
                                has_country_flag = acot_phanon_tech_stage_one_enabled
                            }
                        }
                    }
                    planet_event = {
                        id = acot_pmc.12
                    }
                }
                if = {
                    limit = {
                        planet_owner = {
                            has_country_flag = acot_phanon_tech_stage_one_enabled
                            NOT = {
                                has_country_flag = acot_phanon_tech_stage_two_enabled
                            }
                        }
                    }
                    planet_event = {
                        id = acot_pmc.13
                    }
                }
                if = {
                    limit = {
                        planet_owner = {
                            has_country_flag = acot_phanon_tech_stage_two_enabled
                            NOT = {
                                has_country_flag = acot_phanon_tech_stage_three_enabled
                            }
                        }
                    }
                    planet_event = {
                        id = acot_pmc.14
                    }
                }
                if = {
                    limit = {
                        planet_owner = {
                            has_country_flag = acot_phanon_tech_stage_three_enabled
                            NOT = {
                                has_country_flag = acot_phanon_tech_stage_four_enabled
                            }
                        }
                    }
                    planet_event = {
                        id = acot_pmc.15
                    }
                }
            }
            else = {
                planet_event = {
                    id = acot_pmc.16
                }
            }
            create_vault_defense_drones = yes
        }
    }
    ai_weight = {
        weight = 10000
    }
}

decision_enigmatic_mass_recruitment = {
    enactment_time = 120

    resources = {
        category = armies
        cost = {
            energy = 100
            alloys = 2500

        }
    }

    potential = {
        exists = owner
        owner = {
            is_ai = yes
            has_technology = tech_dark_matter_power_core_ae
            is_at_war = yes
        }
    }
    abort_trigger = {
        exists = owner
        owner = {
            is_at_war = no
        }
    }

    effect = {
        #custom_tooltip = decision_maginot_mass_recruitment_effect
        hidden_effect = {
            if = {
                limit = { exists = planet_owner }
                from = {
                    set_country_flag = acot_decision_army_haver
                }
                if = {
                    limit = { planet_owner = { is_machine_empire = no } }
                    while = {
                        set_update_modifiers_batch = begin
                        count = 10
                        create_army = {
                            owner = planet_owner
                            species = planet_owner.species
                            type = acot_special_force_ae
                            effect = {
                                set_army_flag = acot_war_army
                            }
                        }
                        set_update_modifiers_batch = end
                    }
                }
                else_if = {
                    limit = { planet_owner = { is_hive_empire = yes } }
                    while = {
                        count = 10
                        create_army = {
                            owner = planet_owner
                            species = planet_owner.species
                            type = acot_hive_heavy_ae
                            effect = {
                                set_army_flag = acot_war_army
                            }
                        }
                    }
                }
                else_if = {
                    limit = { planet_owner = { is_machine_empire = yes } }
                    while = {
                        count = 10
                        create_army = {
                            owner = planet_owner
                            species = planet_owner.species
                            type = acot_army_machine_heavy_ae
                            effect = {
                                set_army_flag = acot_war_army
                            }
                        }
                    }
                }
            }
        }
    }

    ai_weight = {
        # this is for player  (in)convenience
        weight = 300
    }
}
